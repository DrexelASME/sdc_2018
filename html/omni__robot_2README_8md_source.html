83 ## Motor Class
84 
85 This class is used for motors that have an encoder that works with the Encoder class and has an optional current
86 sensor.
87 
88 ```cpp
89 #ifndef MOTOR_H_
90 #define MOTOR_H_
91 
92 #include "Encoder.h" // Encoder class and MOTOR_DIR_CW MOTOR_DIR_CCW
93 #include "Sensor.h" // Sensor class
94 
95 #define MOTOR_TYPE_DC 1
96 #define MOTOR_TYPE_BLDC 2
97 #define MOTOR_TYPE_SERVO 3
98 
99 struct MotorState {
100  int duty_cycle; // value between 0 and 100
101  bool direction; // either MOTOR_DIR_CW or MOTOR_DIR_CCW (from Encoder.h)
102 };
103 struct MotorPins {
104  int PIN_PWM;
105  int PIN_A;
106  int PIN_B;
107 };
108 
109 class Motor {
110 public:
111  void setupMotor();
112  double getCurrentUsage();
113  double getMotorPosition();
114 
115 protected:
116  MotorState state;
117  MotorPins pins;
118  static int TOTAL_MOTORS;
119 
120 private:
121  Encoder encoder;
122  Sensor current;
123 };
124 
125 #endif /* MOTOR_H_ */
126 ```

178 ## Sensor Class
179 
180 This is a generalized sensor class for the current, IR TOF (time-of-flight), and compass sensors.
181 
182 ```cpp
183 #ifndef SENSOR_H_
184 #define SENSOR_H_
185 
186 #define SENSOR_TYPE_CURRENT 1
187 #define SENSOR_TYPE_IR_DIGITAL 2
188 #define SENSOR_TYPE_COMPASS 3
189 
190 #define SENSOR_COMM_TYPE_DIGITAL 1
191 #define SENSOR_COMM_TYPE_PULSE 2 // PWM or PPM
192 #define SENSOR_COMM_TYPE_I2C 3
193 
194 struct SensorState {
195  volatile long int timestamp; // time of last update
196  volatile double data;
197 };
198 struct SensorPins {
199  int PIN_DATA;
200  int PIN_CLOCK;
201  int PIN_INTERRUPT;
202 };
203 
204 class Sensor {
205 public:
206  void setupDataInterrupt();
207 
208 protected:
209  SensorState state;
210  SensorPins pins;
211  static int TOTAL_SENSORS;
212 
213  int COMM_TYPE;
214  int DATA_SCALAR;
215  bool INTERRUPTABLE; // whether data can be updated on interrupt or not
216 };
217 
218 #endif /* SENSOR_H_ */
219 ```

226 ## 6 Channel RC Controller Class
227 
228 This class is used to take the stick inputs from an RC controller through a 6CH receiver.
229 
230 ```cpp
231 #ifndef RC_CONTROLLER_6CH_H_
232 #define RC_CONTROLLER_6CH_H_
233 
234 #define DEFAULT_CH_VALUE_MIN 1000 // min PPM signal in microseconds
235 #define DEFAULT_CH_VALUE_MAX 2000 // max PPM signal in microseconds
236 
237 #define CH_MAP_TYPE_BINARY 1 // maps channel values to 0 to 1
238 #define CH_MAP_TYPE_UNIDIRECTIONAL 2 // maps channel values to 0 to 100
239 #define CH_MAP_TYPE_BIDIRECTIONAL 3 // maps channel values to -100 to 100
240 
241 struct RCController6CHState {
242  volatile long int timestamp; // time of last update
243  volatile int value_ch_1;
244  volatile int value_ch_2;
245  volatile int value_ch_3;
246  volatile int value_ch_4;
247  volatile int value_ch_5;
248  volatile int value_ch_6;
249  int *values[] = {&value_ch_1, &value_ch_2, &value_ch_3, &value_ch_4,
250  &value_ch_5, &value_ch_6}; // iterate through channel values
251 };
252 struct RCController6CHPins {
253  int PIN_CH_1;
254  int PIN_CH_2;
255  int PIN_CH_3;
256  int PIN_CH_4;
257  int PIN_CH_5;
258  int PIN_CH_6;
259  int *PINS[] = {&PIN_CH_1, &PIN_CH_2, &PIN_CH_3, &PIN_CH_4, &PIN_CH_5, &PIN_CH_6}; // iterate through pins
260 };
261 
262 class RCController6CH {
263 public:
264  void setupChannel(int channel, int map_type, int threshold);
265  void setupChannel(int channel, int map_type, int min_value, int max_value);
266 
267 protected:
268  RCController6CHState state;
269  RCController6CHPins pins;
270  int CH_UPDATE_HZ; // frequency at which the controller input model is executed
271 
272 private:
273  void runControllerInputModel(); // defined in robot class
274  void setupControllerInputTimerInterupt(); // interrupt to update controller input model
275  void setupChannelInterrupts();
276 };
277 
278 #endif /* RC_CONTROLLER_6CH_H_ */
279 ```
